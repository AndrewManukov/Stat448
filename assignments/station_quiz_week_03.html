data ccso;
 set ccso;
 if daysinjail>30 then monthbin=1;
  else monthbin=0;
 if race="White" then whitebin=1;
  else whitebin=0;
run;
proc freq data=ccso;
 tables monthbin*whitebin / chisq;
 ods select chisq;
run;
proc corr data=ccso spearman;
 var monthbin whitebin;
 ods select spearmancorr;
run;
